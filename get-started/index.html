# 快速上手

引入必要资源

import {Button} from "tears-ui";
import "tears-ui/dist/index.css";

注册组件

export default {
  name: "ComponentName",
  components: {
    "g-button": Button,
  },
}

使用

<template>
  <div>
    <g-button>默认按钮</g-button>
  </div>
</template>
